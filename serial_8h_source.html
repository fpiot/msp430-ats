667 hBSP430halSERIAL hBSP430serialLookup (tBSP430periphHandle periph)
668 {
669  hBSP430halSERIAL rv = NULL;
670 #if configBSP430_SERIAL_USE_USCI - 0
671  if (NULL == rv) {
672  rv = hBSP430usciLookup(periph);
673  }
674 #endif /* configBSP430_SERIAL_USE_USCI */
675 #if configBSP430_SERIAL_USE_USCI5 - 0
676  if (NULL == rv) {
677  rv = xBSP430usci5Lookup(periph);
678  }
679 #endif /* configBSP430_SERIAL_USE_USCI5 */
680 #if configBSP430_SERIAL_USE_EUSCI - 0
681  if (NULL == rv) {
682  rv = hBSP430eusciLookup(periph);
683  }
684 #endif /* configBSP430_SERIAL_USE_EUSCI */
685  return rv;
686 }
