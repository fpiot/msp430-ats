711 static BSP430_CORE_INLINE
712 hBSP430halSERIAL hBSP430serialLookup (tBSP430periphHandle periph)
713 {
714  hBSP430halSERIAL rv = NULL;
715 #if configBSP430_SERIAL_USE_USCI - 0
716  if (NULL == rv) {
717  rv = hBSP430usciLookup(periph);
718  }
719 #endif /* configBSP430_SERIAL_USE_USCI */
720 #if configBSP430_SERIAL_USE_USCI5 - 0
721  if (NULL == rv) {
722  rv = hBSP430usci5Lookup(periph);
723  }
724 #endif /* configBSP430_SERIAL_USE_USCI5 */
725 #if configBSP430_SERIAL_USE_EUSCI - 0
726  if (NULL == rv) {
727  rv = hBSP430eusciLookup(periph);
728  }
729 #endif /* configBSP430_SERIAL_USE_EUSCI */
730  return rv;
731 }
